  # Получаем идентификатор искомого оборудования
  ?b prop:type type:o .
  ?b prop:title ?title FILTER (?title = "ШВ") .
  # Проверяем в цикле все свзяки по свойству (powered)
  ?b (prop:powered+) ?c .
  # Выодим название объектов полученных после цикла
  ?c prop:title ?cc .
  # Выводим название цеха расположения объектов
  ?c prop:place ?ss .
  ?ss prop:title ?tt
  
}
